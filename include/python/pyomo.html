    Pyomo支持differential algebraic equations(DAE), 把连续空间离散化(Discretize), 一般有两种方式: 1) backward finite difference; 2) collocation transformation

    • backward finite difference: 比如dx/dt=f(x,t),x(t0)=x0dx/dt = f(x,t), x(t_0) = x_0dx/dt=f(x,t),x(t0​)=x0​被离散化成(hhh: step size)

    xk+1=xk+h∗f(tk+1,xk+1)tk+1=tk+h,x(t0+kh)=xk\begin{aligned} x_{k + 1} &= x_{k} + h * f(t_{k + 1}, x_{k + 1}) \\ t_{k + 1} &= t_{k} + h, \quad x(t_0 + kh) = x_{k} \end{aligned}xk+1​tk+1​​=xk​+h∗f(tk+1​,xk+1​)=tk​+h,x(t0​+kh)=xk​​

    • 离散化参数(nfe): number of finite elements, 默认值是101010
    • 离散化参数(wrt): with respect to, 哪个参数需要被离散化, 例如wrt=model.time
    • collocation transformation: 目前两种算法: 1) gauss-radau root; 2) gauss-legendre root; 还可以通过reduce_collocation_points来约束"collocation point per finite element", 比如右边这个是添加了

set S_SI := 1 2 3 4;

param N := 300.000000;

param CI default 0.0 :=
    1 6.000000
    2 3.000000
;
"""

model = AbstractModel()
model.S_SI = Set(ordered=True)                #i = {1, 2, ..., T}
model.CI = Param(model.S_SI)                  #reported case each day
model.N = Param()                             #total population
model.I = Var(model.S_SI, bounds=(0,model.N), initialize=1)
model.S = Var(model.S_SI, bounds=(0,model.N), initialize=300)
model.beta = Var(bounds=(0.05, 70))
model.alpha = Var(bounds=(0.5, 1.5))
model.eps_I = Var(model.S_SI, initialize=0.0)

def _objective(model):
    return sum((model.eps_I[i])**2 for i in model.S_SI)
model.objective = Objective(rule=_objective, sense=minimize)

def _InfDynamics(model, i):
    if i != 1:
        return model.I[i] == (model.beta * model.S[i-1] * model.I[i-1]**model.alpha)/model.N
    return Constraint.Skip
model.InfDynamics = Constraint(model.S_SI, rule=_InfDynamics)

def _SusDynamics(model, i):
    if i != 1:
        return model.S[i] == model.S[i-1] - model.I[i]
    return Constraint.Skip
model.SusDynamics = Constraint(model.S_SI, rule=_SusDynamics)

def _Data(model, i):
    return model.CI[i] == model.I[i]+model.eps_I[i]
model.Data = Constraint(model.S_SI, rule=_Data)

data = DataPortal()
data.load(filename="./data/disease.dat", model=model)
instance = model.create_instance(data)
result = opt.solve(instance); instance.pprint()

    from pyomo.environ import *
# P: 机长, C: 副机长, V: 航班
model = AbstractModel()
model.P =Set(initialize=crew['Pilots'].dropna().tolist())
model.C = Set(initialize=crew['Co-pilots'].dropna().tolist())
model.V = Set(initialize=flights['Flights'].tolist())
# t: 每个航班所需飞行时间(字典), I: 相互冲突的航班
model.t = Param(model.V, initialize={r['Flights']: r['Time'] for i, r in flights.iterrows()})
model.I = Param(model.V, model.V, rule=lambda model, v1, v2: 1 if v1 in flights[v2].tolist() else 0)
# f, g: 单位时间机长和副机长的工资
get_cost = lambda m,p,v: costs.at[p, v]
model.f = Param(model.P, model.V, rule=get_cost)
model.g = Param(model.C, model.V, rule=get_cost)
model.q = Param(initialize=10)
# x, y: 决策变量, 每一个机长和副机长, 给安排哪些航班
model.x = Var(model.P, model.V, within=Binary)
model.y = Var(model.C, model.V, within=Binary)
# z: 目标函数, 最小化总的薪酬
def get_object(m):
    pf = lambda v: sum(m.f[p, v] * m.x[p, v] for p in m.P)
    cf = lambda v: sum(m.g[c, v] * m.y[c, v] for c in m.C)
    return sum(m.t[v] * (pf(v) + cf(v)) for v in m.V)
model.z = Objective(rule=get_object, sense=minimize)
# c1: 每一航班只有一名机长和副机长
c1_p = lambda m,v: sum(m.x[p,v] for p in m.P) == 1
c1_c = lambda m,v: sum(m.y[c,v] for c in m.C) == 1

model.n = Set(initialize=range(n))

model.D = Set(initialize=drones)          # set of drones
model.r = Param(model.D, rule=lambda _: random.randrange(2,4)) 
# x, y: 决策变量, x代表无人机所在的区域位置(避免相撞), y代表某区域是否被覆盖
model.x = Var(model.D, model.M, within=Binary)
model.y = Var(model.M, within=Binary)
# z: 目标函数, 最大化覆盖率
def get_object(m):
    return sum(m.y[i, j] for (i, j) in m.M)
