def make_model(metrics = METRICS, output_bias=None):
    if output_bias is not None:
        output_bias = tf.keras.initializers.Constant(output_bias)
    model = keras.Sequential([
        keras.layers.Dense(16, activation='relu', input_shape=(train_features.shape[-1],)),
        keras.layers.Dropout(0.5),
        keras.layers.Dense(1, activation='sigmoid', bias_initializer=output_bias),
    ])
    model.compile(
        optimizer=keras.optimizers.Adam(lr=1e-3),
        loss=keras.losses.BinaryCrossentropy(),
        metrics=metrics)
    return model

Early Stopping

  • 回答整个训练应该遍历所有样本多少次的问题(一遍为一个epoch), 如果遍历过少, 则训练可能欠拟合, 遍历过多, 则可能过拟合
  • 和正则化差不多, 防止参数过大让分类不稳定

参数包括

  • monitor: 包括val_acc, val_auc等等
  • min_delta: 多少进步算作improvement, 比如0.03%
  • patience: 能够容忍多少epoch内都没有improvement, 配合min_delta使用
  • mode: objective function, 配合monitor, 如果是acc, 就选择min
# 容忍度是10, 标准是验证集上的AUC
early_stopping = tf.keras.callbacks.EarlyStopping(
    monitor='val_auc', 
    patience=10,
    mode='max',
    restore_best_weights=True, verbose=1)

Imbalance Classification

  • 给初始输出层加一个偏置, 假设初始神经网络w⋅x=0w\cdot x = 0w⋅x=0, 那么

P(y=1∣x)=(1+exp⁡(−w⋅x+b))−1P(y=1 \vert x) = (1 + \exp(-w\cdot x + b))^{-1} P(y=1∣x)=(1+exp(−w⋅x+b))−1

  • 如果b=0⇒P(y=1∣x)=12b = 0 \Rightarrow P(y=1 \vert x) = \frac{1}{2}b=0⇒P(y=1∣x)=21​, 符合我们对样本均衡数据的分类期望
  • 如果b=ln⁡n+n−⇒P(y=1∣x)=n+n++n−b = \ln\frac{n^{+}}{n^{-}} \Rightarrow P(y=1\vert x) = \frac{n^{+}}{n^{+}+n^{-}}b=lnn−n+​⇒P(y=1∣x)=n++n−n+​, 这样模型一开始会倾向于样本更多的类别, 在训练过程中会对较少样本的错误分类进行修正, 更专注于样本较少样本的差异化特征

在当前这个信用卡欺诈的例子里, 正例概率只有0.17%, 加了这个偏执, 模型更容易把正样本预测为负, 因此会更加专注去修正错误分类的正样本, 即提高recall(真正0.17%的正例中你猜对了多少)

  • 在计算损失函数的时候, 给样本少的类别赋予更大的权重(class weight), 让模型不要错分较少样本的类别, 在本例中, 赋予正例更高的权重
neg, pos = np.bincount(df['Class']); total = pos + neg
cls_weight = {0: (1/neg)*(total)/2.0, 1: (1/pos)*(total)/2.0}
  • 在训练时, 对样本少的类别进行过采样, 在本例中, 重复采样正例, 和赋予权重效果应该差不多吧

from sklearn.metrics import roc_curve

def plot_roc(name, labels, predictions, ax, **kwargs):
    fp, tp, _ = roc_curve(labels, predictions)
    ax.plot(100*fp, 100*tp, label=name, linewidth=2, **kwargs)
    ax.set_xlim([-0.5,20]); ax.set_ylim([80,100.5])
    ax.grid(True)
    ax.set_aspect('equal')

def plot_metrics(histories, titles, metrics, labels):
    fig, axs = plt.subplots(2, 2, figsize=(12, 8))
    fig.subplots_adjust(hspace=0.1,wspace=0.1)
    for i, row in enumerate(axs):
        for j, col in enumerate(row):
            xlabel, ylabel = titles[i][j]; f1, f2 = metrics[i][j]
            if i == 0 and j == 0: col.set_ylabel(ylabel)
            if i == 1 and j == 0: col.set(xlabel=xlabel, ylabel=ylabel)
            if i == 1 and j == 1: col.set_xlabel(xlabel)
            col.grid(True)
            yfn = lambda x: np.array(x[f1])/(np.array(x[f1])+np.array(x[f2]))
            for k, history in enumerate(histories):
                col.plot(history.epoch, yfn(history.history), label=labels[k])
    axs[1][1].legend(loc='lower right', prop={'size':12})
    fig.suptitle('Classification Performance During Training of Each Algorithm')
    fig.tight_layout()
    fig.subplots_adjust(top=0.95)
plot_metrics(histories, titles, metrics, labels)
