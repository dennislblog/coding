def send(self, is_buy, qty, price, uid, is_mine=False, timestamp=datetime.now()):
    """ 
        首先如果订单不是历史订单, 价格就要相应作出调整(考虑到市场影响)
        然后看是否价差相交, 可以直接匹配交易还是加到订单薄的某个位置
    """
    if not is_mine:
        if self.market_impact >= 1:
            nticks = min(int(self.resilience*self.market_impact),
                         self.max_impact)
            price = self.get_new_price(price=price, n_moves=nticks)
        elif self.market_impact <= -1:
            nticks = max(int(self.resilience*self.market_impact),
                         -1 * self.max_impact)
            price = self.get_new_price(price=price, n_moves=nticks)   
    else:
        self.n_my_orders += 1; self.my_cumvol_sent += qty

    neword = Order(uid, is_buy, qty, price, timestamp)
    self._orders.update({uid: neword})
    while (neword.leavesqty > 0):
        if self._is_aggressive(neword):  self._sweep_best_price(neword)
        else:  self._bids.add(neword) or self._asks.add(neword)

latency=mdlatency+algolatency+oblatency\mathbf{latency} = \mathbf{md latency} + \mathbf{algo latency} + \mathbf{ob latency} latency=mdlatency+algolatency+oblatency

  • md_latency, 查询最优价格花费的时间
  • algo_latency, 算法执行和提交订单的时间
  • ob_latency, 物理延迟, 包括路由带宽和离交易所的距离等

# 处理订单

def tick(self):
    """
    如果我的订单在下一个历史订单之前到达, 先处理我的订单
        vol_in_queue: 队列里有多少交易量待执行?
    否则直接
    """
    oborder = self.hist_orders[self.ob_idx]   # 下一个历史订单
    if self.my_queue[0].timestamp < oborder[self.col_idx['timestamp']]:
        my_order = self.my_queue.popleft()  # 如果my_queue不为空的话
        self._send_to_orderbook(my_order, is_mine=True)
        return
    self._send_to_orderbook(oborder, is_mine=False)

def _send_to_orderbook(self, order, is_mine):
    """发送新订单/修改/取消订单
       
        首先看是否还在模拟器结束时间之前, 更新ob_time
        然后提交不同订单行为给订单薄
    """
    ord_type, timestamp = order[self.col_idx['ordtype']], order[self.col_idx['timestamp']]
    if ord_type == "new":
        self.ob.send(is_buy=order[self.col_idx['is_buy']], qty=order[self.col_idx['qty']], price=order[self.col_idx['price']],  uid=order[self.col_idx['uid']], is_mine=is_mine, timestamp=timestamp)
    elif ord_type == "cancel":
        self.ob.cancel(uid=order[self.col_idx['uid']])
    elif ord_type == "modif":
        self.ob.modif(uid=order[self.col_idx['uid']], qty_down=order[self.col_idx['qty']])
